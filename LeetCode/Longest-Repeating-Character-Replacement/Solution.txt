15class Solution {
16public:
17    int characterReplacement(string s, int k) {
18        int n = s.length(), maxFreq = 0, left = 0, right = 0;
19        int maxLength = 0;
20        vector<int> hash(26, 0);
21        while (right < n) {
22            hash[s[right] - 'A']++; // update the freq
23            maxFreq = max(maxFreq, hash[s[right] - 'A']);
24            // check if current window is even valid or not
25            if ((right - left + 1) - maxFreq > k) {
26                hash[s[left] - 'A']--;
27                left++;
28            }
29            // keep in mind that we never update the freq back again
30            // bcs changing the freq is not dependent on the answer
31            if ((right - left + 1) - maxFreq <= k) {
32                maxLength = max(maxLength, (right - left + 1));
33            }
34            right++;
35        }
36        return maxLength;
37    }
38};
